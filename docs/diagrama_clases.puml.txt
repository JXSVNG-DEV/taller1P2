package "modelo.interfaces" {
    interface Consultable {
        + obtenerResumen(): String
        + estaActivo(): boolean
        + obtenerTipo(): String
    }
    interface Transaccionable {
        + depositar(monto: double): void
        + retirar(monto: double): void
        + calcularComision(monto: double): double
        + consultarSaldo(): double
    }
    interface Auditable {
        + obtenerFechaCreacion(): LocalDateTime
        + obtenerUltimaModificacion(): LocalDateTime
        + obtenerUsuarioModificacion(): String
        + registrarModificacion(usuario: String): void
    }
    interface Notificable {
        + notificar(mensaje: String): void
        + obtenerContacto(): String
        + aceptaNotificaciones(): boolean
    }
}

package "modelo.abstractas" {
    abstract class Persona {
        - id: String
        - nombre: String
        - apellido: String
        - fechaNacimiento: LocalDate
        - email: String
        + getNombreCompleto(): String
        + {abstract} calcularEdad(): int
        + {abstract} obtenerTipo(): String
        + {abstract} obtenerDocumentoIdentidad(): String
    }
    abstract class Empleado {
        - legajo: String
        - fechaContratacion: LocalDate
        - salarioBase: double
        - activo: boolean
        + calcularAntigüedad(): int
        + {abstract} calcularSalario(): double
        + {abstract} calcularBono(): double
    }
    abstract class Cuenta {
        - numeroCuenta: String
        - saldo: double
        - bloqueada: boolean
        - historial: Transaccion[20]
        + verificarBloqueada(): void
        + agregarAlHistorial(t: Transaccion): void
        + getHistorial(): Transaccion[]
        + {abstract} calcularInteres(): double
        + {abstract} getLimiteRetiro(): double
        + {abstract} getTipoCuenta(): String
    }

    Persona <|-- Empleado
}

    GerenteSucursal ..|> Auditable
}

package "modelo.banco" {
    class Banco {
        - nombre: String
        - empleados: Empleado[50]

        - cuentas: Cuenta[500]
        + registrarCliente(c: Persona): void
        + registrarEmpleado(e: Empleado): void
        + abrirCuenta(idCliente: String, c: Cuenta): void
        + buscarCliente(id: String): Persona
        + calcularNominaTotal(): double
        + calcularInteresesMensuales(): void
    }
    class Transaccion {
        - id: String
        - monto: double
        - estado: EstadoTransaccion
        - fecha: LocalDateTime
        - descripcion: String
        + cambiarEstado(nuevo: EstadoTransaccion): void
        + generarComprobante(): String
    }

    Banco ..|> Auditable
    Banco "1" o-- "0..200" Persona
    Banco "1" o-- "0..50" Empleado

    Transaccion --> EstadoTransaccion
}
